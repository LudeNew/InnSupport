git config --global user.name "Your name"
git config --global user.email "Your email"
git init

ssh-keygen - генерация SSH ключа
cat ~/.ssh/... - изъять корректный и правильный ssh ключ из файла
git remote add origin (ссылка для связи с gitlab, без скобок) - связь с удаленным репозиторием


git status
git add "" (имя файла с расширением, без скобок) - добавление файла в отслеживаемые
git rm (имя файла с расширением, без скобок) - удаление файла из отслеживаемых, так же можно удалить файл из репозитория


git add .
git commit -m "Initial commit" - инициализационный (первый) коммит
git commit -m "" - Коммит с описанием измеений
git push -u origin (имя ветки, по умолчанию - master, без скобок)

git branch (имя создаваемой ветки, без скобок) - Создание доп. ветки репозитория
git checkout -b (имя создаваемой ветки, без скобок) - Создание доп. ветки репозитория, с моментальным переключением на нее
git checkout (имя ветки, на которую переключаетесь, без скобок) - переключение ветки
git merge (имя ветки, с которой хотим "влить" в текущую, без скобок) - внести изменения из одной ветки, в другую
git clone - актуализация репозитория, скачивание репозитория


git fetch - обновление веток
git log - Информация о коммитах
git diff - Иноформация о изменениях в локальном репозитории после последнего коммита
git remote (git remote -v) - проверка связи с удаленным репозиторием
touch .gitignore - создание файла gitignore
git add *
git remote add github (ссылка для связи с github, без скобок) - привязка второго репозитория, как пример github
